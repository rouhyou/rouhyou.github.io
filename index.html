MARGIN), 1) # Horizontal lines pygame.draw.line(screen, LINE_COLOR, (MARGIN, MARGIN + i * GRID_SIZE), (WINDOW_SIZE - MARGIN, MARGIN + i * GRID_SIZE), 1) # Draw star points (hoshi) star_points = [3, 9, 15] # For 19x19 board for i in star_points: for j in star_points: pygame.draw.circle(screen, BLACK, (MARGIN + i * GRID_SIZE, MARGIN + j * GRID_SIZE), 4) def draw_stones(): for i in range(BOARD_SIZE): for j in range(BOARD_SIZE): if board[i][j] == 1: # Black stone pygame.draw.circle(screen, BLACK, (MARGIN + i * GRID_SIZE, MARGIN + j * GRID_SIZE), STONE_RADIUS) elif board[i][j] == 2: # White stone pygame.draw.circle(screen, WHITE, (MARGIN + i * GRID_SIZE, MARGIN + j * GRID_SIZE), STONE_RADIUS) pygame.draw.circle(screen, BLACK, (MARGIN + i * GRID_SIZE, MARGIN + j * GRID_SIZE), STONE_RADIUS, 1) # Outline for white stones def get_board_position(mouse_pos): x, y = mouse_pos # Convert screen coordinates to board coordinates board_x = round((x -
